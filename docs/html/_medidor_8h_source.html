72/**
73 * @class Medidor
74 * @brief Se encarga de la adquisición y procesamiento de datos ambientales.
75 * * Esta clase encapsula tanto las lecturas de los pines analógicos como
76 * el procesamiento matemático para convertir voltajes en unidades físicas.
77 */
78class Medidor {
79
80private:
81 float _Vref_base = 0.0f; ///< Valor de calibración inicial de VREF.
82
83 /**
84 * @brief Lee el voltaje de un pin analógico promediando varias muestras.
85 * @param pin Pin analógico a leer.
86 * @param nAvg Número de muestras para el promedio.
87 * @return Voltaje calculado en Voltios.
88 */
89 float leerVolt(int pin, int nAvg = 10) {
90 long acc = 0;
91 for (int i=0; i<nAvg; ++i) { acc += analogRead(pin); delay(2); }
92 const float raw = (float)acc / nAvg;
93 const float fullScale = (float)((1 << O3_ADC_BITS) - 1);
94 return (raw * O3_VDD) / fullScale;
95 }
96
97public:

99 /**
100 * @brief Constructor por defecto.

102 Medidor( ) {
103 }

105 /**
106 * @brief Calibra el medidor obteniendo el voltaje de referencia inicial.
107 * @param nAvg Número de muestras para la calibración (por defecto 50).
108 */
109 void iniciarMedidor(int nAvg = 50) {
110 _Vref_base = leerVolt(O3_PIN_VREF, nAvg);
111 }
112
113 /**
114 * @brief Calcula el porcentaje de carga de la batería real.
115 * @details Utiliza un divisor de tensión 2:1 en el pin A6.
116 * @return Porcentaje de batería (0-100).
117 */
118 int medirBateria() {
119 const int nAvg = 10;
120 long rawAcc = 0;
121 for (int i = 0; i < nAvg; ++i) {
122 rawAcc += analogRead(PIN_A6);
123 delay(1);
124 }
125 const float rawAvg = (float)rawAcc / nAvg;
126 const float fullScale = (float)((1 << ADC_BITS) - 1);
127 float measuredVolts = (rawAvg * VDD) / fullScale;
128
129 // Multiplicar por 2 por el divisor de voltaje del hardware
130 float battVolts = measuredVolts * 2.0f;
131
132 float battRange = BATT_MAX_VOLTS - BATT_MIN_VOLTS;
133 float percent = ((battVolts - BATT_MIN_VOLTS) / battRange) * 100.0f;
134
135 if (percent > 100.0f) percent = 100.0f;
136 if (percent < 0.0f) percent = 0.0f;
137
138 return (int)percent;
139 }

176 /**
177 * @brief Obtiene una medida de Ozono de los datos simulados.
178 * @return Valor de ozono en ppm (float).
179 */
180 float medirPPMSimulado() {
181 int indiceAleatorio = random(0, NUM_O3_VALORES);
182 float valorSimuladoRaw = (float)O3_SIMULADO[indiceAleatorio];
183 return valorSimuladoRaw / 1000.0f;
184 }
185
186 /**
187 * @brief Realiza la medición real de Ozono (O3) en ppm.
188 * @details Calcula la diferencia entre Vgas y Vref, convierte a corriente y
189 * aplica las constantes de sensibilidad y corrección (slope/offset).
190 * @return Concentración de ozono corregida en ppm.
191 */
192 float medirPPM() {
193 float Vg = leerVolt(O3_PIN_VGAS, 10);
194 float deltaV = Vg - _Vref_base;
195
196 float denominador = GAIN_TIA * SENSIBILIDAD_SENSOR * 1e-6f;
197
198 float ppm_bruto = 0.0f;
199 if (denominador != 0.0f) {
200 ppm_bruto = deltaV / denominador;
201 }
202
203 ppm_bruto = fabsf(ppm_bruto);
204
205 // Ajuste lineal final
206 float ppm_corregido = (ppm_bruto * CORRECCION_SLOPE) + CORRECCION_OFFSET;
207
208 if (ppm_corregido < 0.0f) {
209 ppm_corregido = 0.0f;
210 }
211
212 return ppm_corregido;
213 }
