93 /**
94 * @brief Configura y emite un anuncio estándar iBeacon.
95 * @param beaconUUID Puntero al array de 16 bytes del UUID.
96 * @param major Valor Major del iBeacon.
97 * @param minor Valor Minor del iBeacon.
98 * @param rssi Valor de RSSI calibrado a 1 metro.
99 */
100 void emitirAnuncioIBeacon( uint8_t * beaconUUID, int16_t major, int16_t minor, uint8_t rssi ) {
101 (*this).detenerAnuncio();
102
103 BLEBeacon elBeacon( beaconUUID, major, minor, rssi );
104 elBeacon.setManufacturer( (*this).fabricanteID );
105
106 Bluefruit.setTxPower( (*this).txPower );
107 Bluefruit.setName( (*this).nombreEmisora );
108 Bluefruit.ScanResponse.addName();
109
110 Bluefruit.Advertising.setBeacon( elBeacon );
111 Bluefruit.Advertising.restartOnDisconnect(true);
112 Bluefruit.Advertising.setInterval(100, 100);
113
114 Bluefruit.Advertising.start( 0 );
115 }

117 /**
118 * @brief Emite un anuncio iBeacon con una carga de datos personalizada.
119 * @param carga Puntero a los datos (máximo 21 bytes).
120 * @param tamanyoCarga Tamaño de los datos a emitir.
121 */
122 void emitirAnuncioIBeaconLibre( const char * carga, const uint8_t tamanyoCarga ) {
123 (*this).detenerAnuncio();
124
125 Bluefruit.Advertising.clearData();
126 Bluefruit.ScanResponse.clearData();
127
128 Bluefruit.setName( (*this).nombreEmisora );
129 Bluefruit.ScanResponse.addName();
130 Bluefruit.Advertising.addFlags(BLE_GAP_ADV_FLAGS_LE_ONLY_GENERAL_DISC_MODE);
131
132 uint8_t restoPrefijoYCarga[4+21] = {
133 0x4c, 0x00, // companyID
134 0x02, // ibeacon type
135 21, // ibeacon length (dec=21)
136 '-','-','-','-','-','-','-','-','-','-','-','-','-','-','-','-','-','-','-','-','-'
137 };
138
139 memcpy( &restoPrefijoYCarga[4], &carga[0], ( tamanyoCarga > 21 ? 21 : tamanyoCarga ) );
140
141 Bluefruit.Advertising.addData( BLE_GAP_AD_TYPE_MANUFACTURER_SPECIFIC_DATA,
142 &restoPrefijoYCarga[0],
143 4+21 );
144
145 Bluefruit.Advertising.restartOnDisconnect(true);
146 Bluefruit.Advertising.setInterval(100, 100);
147 Bluefruit.Advertising.setFastTimeout( 1 );
148 Bluefruit.Advertising.start( 0 );
149
150 Globales::elPuerto.escribir( "emitiriBeacon libre Bluefruit.Advertising.start( 0 ); \n");
151 }

153 /**
154 * @brief Emite datos múltiples en la carga de fabricante.
155 * @details Útil para enviar tramas de sensores empaquetadas.
156 * @param datos Puntero a los datos.
157 * @param tamanyoDatos Longitud de los datos.
158 */
159 void emitirDatosMultiples(const uint8_t *datos, const uint8_t tamanyoDatos) {
160 (*this).detenerAnuncio();
161
162 Bluefruit.Advertising.clearData();
163 Bluefruit.ScanResponse.clearData();
164
165 Bluefruit.setName( (*this).nombreEmisora );
166 Bluefruit.ScanResponse.addName();
167 Bluefruit.Advertising.addFlags(BLE_GAP_ADV_FLAGS_LE_ONLY_GENERAL_DISC_MODE);
168
169 uint8_t payload[2 + tamanyoDatos];
170 payload[0] = (uint8_t)((*this).fabricanteID & 0xFF);
171 payload[1] = (uint8_t)((*this).fabricanteID >> 8);
172
173 memcpy( &payload[2], datos, tamanyoDatos );
174
175 Bluefruit.Advertising.addData( BLE_GAP_AD_TYPE_MANUFACTURER_SPECIFIC_DATA,
176 &payload[0],
177 2 + tamanyoDatos );
178
179 Bluefruit.Advertising.restartOnDisconnect(true);
180 Bluefruit.Advertising.setInterval(100, 100);
181 Bluefruit.Advertising.setFastTimeout( 1 );
182 Bluefruit.Advertising.start( 0 );
183 }
