14/**
15 * @class Publicador
16 * @brief Gestiona la lógica de empaquetado y emisión de datos de sensores.
17 * * Organiza las lecturas en el formato estándar de iBeacon, utilizando el
18 * campo 'Major' para el ID de la medida y un contador, y el campo 'Minor'
19 * para el valor de la medición.
20 */
21class Publicador {
22
23private:
24 /** * @brief UUID del beacon (16 bytes).

26 */
27 uint8_t beaconUUID[16] = {

29 '-', 'P', 'R', 'O', 'Y', '-', '3', 'A'
30 };

32public:

38 EmisoraBLE laEmisora {
39 "rocio",

41 4

49 /**
50 * @enum MedicionesID
51 * @brief Identificadores numéricos para los distintos tipos de sensores.

53 enum MedicionesID {
54 CO2 = 11, ///< ID para sensores de Dióxido de Carbono.
55 TEMPERATURA = 12, ///< ID para sensores de Temperatura.
56 RUIDO = 13 ///< ID para sensores de Ruido ambiental.
57 };

73 /**
74 * @brief Publica una medición de CO2.
75 * @param valorCO2 Valor de la medición a enviar (se coloca en el campo Minor).
76 * @param contador Valor incremental para diferenciar anuncios.
77 * @param tiempoEspera Milisegundos que estará activo el anuncio antes de detenerse.
78 * @details El campo 'Major' se forma combinando el ID del CO2 (8 bits altos) y el contador (8 bits bajos).
79 */
80 void publicarCO2( int16_t valorCO2, uint8_t contador,
81 long tiempoEspera ) {
82
83 // 1. Calculamos el campo Major (ID + Contador)
84 uint16_t major = (MedicionesID::CO2 << 8) + contador;
85
86 // 2. Emitimos el anuncio iBeacon
87 (*this).laEmisora.emitirAnuncioIBeacon( (*this).beaconUUID,
88 major,
89 valorCO2, // Minor
90 (*this).RSSI
91 );
92
93 // 3. Esperamos y detenemos
94 esperar( tiempoEspera );
95 (*this).laEmisora.detenerAnuncio();
96 }
97
98 /**
99 * @brief Publica una medición de Temperatura.
100 * @param valorTemperatura Valor térmico (se coloca en el campo Minor).
101 * @param contador Valor incremental.
102 * @param tiempoEspera Duración del anuncio en milisegundos.
103 */
104 void publicarTemperatura( int16_t valorTemperatura,
105 uint8_t contador, long tiempoEspera ) {
106
107 uint16_t major = (MedicionesID::TEMPERATURA << 8) + contador;
108
109 (*this).laEmisora.emitirAnuncioIBeacon( (*this).beaconUUID,
110 major,
111 valorTemperatura, // Minor
112 (*this).RSSI
113 );
114
115 esperar( tiempoEspera );
116 (*this).laEmisora.detenerAnuncio();
117 }
