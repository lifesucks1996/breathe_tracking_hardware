16/**
17 * @brief Invierte el orden de los elementos en un array.
18 * @tparam T Tipo de los elementos del array.
19 * @param p Puntero al array.
20 * @param n Número de elementos.
21 * @return Puntero al array modificado.
22 */
23template< typename T >
24T * alReves( T * p, int n ) {
25 T aux;
26 for( int i=0; i < n/2; i++ ) {
27 aux = p[i];
28 p[i] = p[n-i-1];
29 p[n-i-1] = aux;
30 }
31 return p;
32}

34/**
35 * @brief Convierte una cadena de texto a un array de uint8_t en orden inverso.
36 * @details Útil para convertir strings de UUID a formato Little Endian requerido por la pila BLE.
37 * @param pString Cadena de origen (C-string).
38 * @param pUint Puntero al array de destino.
39 * @param tamMax Tamaño máximo a copiar (normalmente 16 para UUIDs).
40 * @return Puntero al array de destino.
41 */
42uint8_t * stringAUint8AlReves( const char * pString, uint8_t * pUint, int tamMax ) {
43 int longitudString = strlen( pString );
44 int longitudCopiar = ( longitudString > tamMax ? tamMax : longitudString );
45 for( int i=0; i<=longitudCopiar-1; i++ ) {
46 pUint[ tamMax-i-1 ] = pString[ i ];
47 }
48 return pUint;
49}
